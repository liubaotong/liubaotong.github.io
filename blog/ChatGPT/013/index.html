    score: 0,

    lines: 0,
    blocks: [],
    currentBlock: null,
    nextBlock: null,
    gameover: false
};

/* 初始化游戏 */
function initGame() {
    // 初始化游戏状态
    gameState = {
        score: 0,
        level: 1,
        lines: 0,
        blocks: [],
        currentBlock: null,
        nextBlock: null,
        gameover: false
    };
    // 初始化游戏区域

    // 随机生成当前方块和下一个方块

    generateCurrentBlock();
}

/* 随机生成下一个方块 */
function generateNextBlock() {
    const index = Math.floor(Math.random() * 3);
    gameState.nextBlock = blocks[index];
}

/* 生成当前方块 */
function generateCurrentBlock() {
    gameState.currentBlock = gameState.nextBlock;
    generateNextBlock();

function drawGameArea() {
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    // 绘制当前方块
    const currentBlock = gameState.currentBlock;
    const blockShape = currentBlock.shape;
    for (let r = 0; r < blockShape.length; r++) {
        for (let c = 0; c < blockShape[r].length; c++) {
            if (blockShape[r][c]) {
                const blockX = (10 - blockShape[r].length) * 32 / 2 + c * 32;
                const blockY = r * 32;
                ctx.fillStyle = currentBlock.color;
                ctx.fillRect(blockX, blockY, 32, 32);
            }
        }
    }
    // 绘制已落下的方块
    for (let i = 0; i < gameState.blocks.length; i++) {
        for (let j = 0; j < gameState.blocks[i].length; j++) {
            if (gameState.blocks[i][j]) {
                const blockX = j * 32;
                const blockY = i * 32;
                ctx.fillStyle = gameState.blocks[i][j];
                ctx.fillRect(blockX, blockY, 32, 32);
            }
        }
    }
}

function gameLoop() {
    drawGameArea();
}

const ctx = canvas.getContext('2d');

/* 棋盘格子大小 */
const gridSize = 30;

/* 棋盘状态 */
let boardState = null;

/* 游戏初始化 */
function initGame() {
    // 初始化棋盘状态
    boardState = [];
    for (let i = 0; i < 15; i++) {
        boardState[i] = Array(15).fill(null);
    }
    // 绘制棋谱
    drawBoard();
}

/* 绘制棋盘 */
function drawBoard() {
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    // 绘制棋盘网格
    ctx.beginPath();
    for (let i = 0; i < 15; i++) {
        ctx.moveTo(gridSize / 2 + i * gridSize, gridSize / 2);
        ctx.lineTo(gridSize / 2 + i * gridSize, gridSize / 2 + gridSize * 14);
        ctx.moveTo(gridSize / 2, gridSize / 2 + i * gridSize);
        ctx.lineTo(gridSize / 2 + gridSize * 14, gridSize / 2 + i * gridSize);
    }
    ctx.stroke();
}

/* 绘制棋子 */
function drawStones(x, y, color) {
    ctx.beginPath();
    ctx.arc(gridSize / 2 + x * gridSize, gridSize / 2 + y * gridSize, gridSize / 2 - 2, 0, 2 * Math.PI);
    ctx.fillStyle = color;
    ctx.fill();
}

/* 棋盘点击事件 */
canvas.addEventListener('click', function(event) {
    const x = event.clientX - canvas.offsetLeft;
    const y = event.clientY - canvas.offsetTop;
    const row = Math.floor(y / gridSize);
    const col = Math.floor(x / gridSize);
    if (boardState[row][col] !== null) {
        return;
    }
    let color = 'black';
    if (boardState.filter(row => row.includes(null)).length % 2 === 0) {
        color = 'white';
    }
    boardState[row][col] = color;
    drawStones(col, row, color);
});

/* 启动游戏 */
initGame();

SVG（Scalable Vector Graphics）是一种使用XML描述2D图形的文件格式。它是一种使用矢量图形（Vector Graphics）方式描述图像的格式，与传统的位图（Bitmap）图像不同，使用SVG可以无损地缩放和改变大小而不会失真。SVG的应用非常广泛，包括图标、线条图、地图、数据可视化等。下面是一个简单的SVG图形的示例：

示例二：在Astro中使用MDX

以下示例代码演示了如何使用Astro在页面中呈现MDX：

import {MDX} from '@astrojs/renderer';

export default AstroComponent({content}) {
  return (
    <div>
      <MDX>{content}</MDX>
    </div>
  )
}

上述代码中使用了 @astrojs/renderer 包中提供的 <MDX> 组件来呈现MDX。这里的 content 是一个字符串，它包含一个MDX文档。 我们将 content 字符串传递给 <MDX> 组件。 当组件被渲染时，MDX会将Markdown文本解析成React组件，并呈现它们。

使用MDX可以创建更具交互性和可定制性的内容，同时还可以保留Markdown的易于使用和阅读的语法。
